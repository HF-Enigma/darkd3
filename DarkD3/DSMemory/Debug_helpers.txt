;------------------------------------------------------------------------------
;  CDSString
;------------------------------------------------------------------------------
ds_utils::CDSString{
	children
	(
		#( 
			[!raw data]:[$c,!],
			[size]: 	[$c.m_size],
			[length]:	[$c.m_length],
			
			#array
			(
				expr: $c.m_pwData[$i],
				size: $c.m_size
			)
		)
    )
	
	preview
	(
		[$c.m_pwData, su]
	)
}
